224void gerer_clic_inventaire(t_case **inventaire, t_case **hotbar, SDL_Event *event, int inventaire_size, int inventaire_cols, int inventaire_rows) {
225 if (event->type != SDL_EVENT_MOUSE_BUTTON_DOWN || event->button.button != SDL_BUTTON_LEFT)
226 return;
227
228 float mx = event->button.x;
229 float my = event->button.y;
230
231 // Calcul position panneau inventaire (même que afficher_inventaire)
232 float panneau_w = inventaire_cols * 70.0f + 20.0f;
233 float panneau_h = inventaire_rows * 70.0f + 60.0f;
234 float panneau_x = (screen_widthf() - panneau_w) / 2.0f;
235 float panneau_y = (screen_heightf() - panneau_h) / 2.0f;
236 float hotbar_x = screen_center_x() - ((HOTBAR_SIZE * 70.0f - 10.0f) * 0.5f);
237 float hotbar_y = screen_heightf() - 80.0f;
238
239 int clic_slot = -1;
240
241 // Vérifier clic sur inventaire
242 for (int row = 0; row < inventaire_rows; row++) {
243 for (int col = 0; col < inventaire_cols; col++) {
244 int i = row * inventaire_cols + col;
245 if (i >= inventaire_size) continue;
246 float sx = panneau_x + 10.0f + col * 70.0f;
247 float sy = panneau_y + 50.0f + row * 70.0f;
248 if (mx >= sx && mx <= sx + 60 && my >= sy && my <= sy + 60) {
249 clic_slot = i;
250 }
251 }
252 }
253
254 // Vérifier clic sur hotbar
255 for (int i = 0; i < HOTBAR_SIZE; i++) {
256 float sx = hotbar_x + i * 70.0f;
257 float sy = hotbar_y;
258 if (mx >= sx && mx <= sx + 60 && my >= sy && my <= sy + 60) {
259 clic_slot = inventaire_size + i;
260 }
261 }
262
263 if (clic_slot == -1) {
264 // Clic en dehors : désélectionner
265 slot_selectionne = -1;
266 return;
267 }
268
269 if (slot_selectionne == -1) {
270 // Aucun slot sélectionné : sélectionner ce slot
271 slot_selectionne = clic_slot;
272 return;
273 }
274
275 if (slot_selectionne == clic_slot) {
276 // Clic sur le même slot : désélectionner
277 slot_selectionne = -1;
278 return;
279 }
280
281 t_case **slot_a = NULL;
282 t_case **slot_b = NULL;
283
284 if (slot_selectionne < inventaire_size)
285 slot_a = &inventaire[slot_selectionne];
286 else
287 slot_a = &hotbar[slot_selectionne - inventaire_size];
288
289 if (clic_slot < inventaire_size)
290 slot_b = &inventaire[clic_slot];
291 else
292 slot_b = &hotbar[clic_slot - inventaire_size];
293
294 t_case *tmp = *slot_a;
295 *slot_a = *slot_b;
296 *slot_b = tmp;
297
298 slot_selectionne = -1;
299}
