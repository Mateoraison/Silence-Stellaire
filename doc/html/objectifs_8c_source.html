73#define AJOUTER(desc_,verif_) do { \
74 obj->objectifs[obj->nb].description = (desc_); \
75 obj->objectifs[obj->nb].valide = false; \
76 obj->objectifs[obj->nb].verifier = (verif_);\
77 obj->nb++; \
78} while(0)
79
80 switch (planete) {
81 case 1:
82 AJOUTER("Ramasser un engrenage", verif_ramasser_engrenage);
83 AJOUTER("Manger de la viande cuite", verif_viande_cuite);
84 AJOUTER("Accumuler 5 pièces", verif_cinq_pieces);
85 AJOUTER("Se soigner à pleine vie", verif_pleine_vie);
86 break;
87 case 2:
88 AJOUTER("Ramasser un engrenage", verif_ramasser_engrenage);
89 AJOUTER("Résoudre le puzzle 1", verif_mastermind_termine);
90 AJOUTER("Résoudre le puzzle 2", verif_simon_termine);
91 AJOUTER("Vaincre FireDemon", verif_boss1_tue);
92 break;
93 case 3:
94 AJOUTER("Trouver l'engrenage dans le labyrinthe", verif_ramasser_engrenage);
95 AJOUTER("Vaincre le Minotaure", verif_boss3_tue);
96 break;
97 default:
98 AJOUTER("Ramasser un engrenage", verif_ramasser_engrenage);
99 AJOUTER("Réparer le vaisseau", verif_reparer_vaisseau);
100 break;
101 }
102#undef AJOUTER
103}

127void objectifs_afficher(t_objectifs *obj, SDL_Renderer *renderer, TTF_Font *font) {
128 if (!obj->visible || !font) return;
129
130 float panneau_x = screen_widthf() - PANNEAU_W - 20.0f;
131 if (panneau_x < 10.0f) panneau_x = 10.0f;
132 float panneau_y = PANNEAU_Y;
133
134 Uint32 maintenant = SDL_GetTicks();
135
136 for (int i = 0; i < obj->nb; i++) {
137 if (!obj->objectifs[i].valide && obj->objectifs[i].verifier) {
138 if (obj->objectifs[i].verifier()) {
139 obj->objectifs[i].valide = true;
140 obj->anim_slot = i;
141 obj->anim_debut = maintenant;
142 }
143 }
144 }
145
146
147 /* Calcul hauteur totale du panneau */
148 float hauteur = PADDING * 2 + 24.0f /* titre */ + PADDING + obj->nb * LIGNE_H;
149
150 /* Assombrir l'ecran pour faire ressortir la liste des objectifs */
151 SDL_SetRenderDrawBlendMode(renderer, SDL_BLENDMODE_BLEND);
152 SDL_SetRenderDrawColor(renderer, 10, 10, 30, 200);
153 SDL_FRect fond = { panneau_x, panneau_y, PANNEAU_W, hauteur };
154 SDL_RenderFillRect(renderer, &fond);
155
156 /* Bordure */
157 SDL_SetRenderDrawColor(renderer, 80, 130, 200, 220);
158 SDL_RenderRect(renderer, &fond);
159
160 /* Titre */
161 SDL_Color blanc = {255, 255, 255, 255};
162 SDL_Surface *s_titre = TTF_RenderText_Solid(font, "Objectifs", 0, blanc);
163 if (s_titre) {
164 SDL_Texture *t = SDL_CreateTextureFromSurface(renderer, s_titre);
165 SDL_FRect dst = {
166 panneau_x + (PANNEAU_W - s_titre->w) / 2.0f,
167 panneau_y + PADDING,
168 (float)s_titre->w, (float)s_titre->h
169 };
170 SDL_RenderTexture(renderer, t, NULL, &dst);
171 SDL_DestroyTexture(t);
172 SDL_DestroySurface(s_titre);
173 }
174
175 /* Séparateur sous le titre */
176 SDL_SetRenderDrawColor(renderer, 80, 130, 200, 150);
177 SDL_FRect sep = { panneau_x + PADDING, panneau_y + PADDING + 22.0f, PANNEAU_W - PADDING * 2, 1.0f };
178 SDL_RenderFillRect(renderer, &sep);
179
180 /* Lignes d'objectifs */
181 float y = panneau_y + PADDING + 24.0f + PADDING;
182
183 for (int i = 0; i < obj->nb; i++) {
184 bool en_anim = (obj->anim_slot == i) &&
185 (maintenant - obj->anim_debut < ANIM_DUREE);
186
187 /* Surbrillance de validation */
188 if (en_anim) {
189 float alpha_f = 1.0f - (float)(maintenant - obj->anim_debut) / (float)ANIM_DUREE;
190 SDL_SetRenderDrawBlendMode(renderer, SDL_BLENDMODE_BLEND);
191 SDL_SetRenderDrawColor(renderer, 50, 220, 80, (Uint8)(alpha_f * 160));
192 SDL_FRect hl = { panneau_x + 2, y - 2, PANNEAU_W - 4, LIGNE_H };
193 SDL_RenderFillRect(renderer, &hl);
194 }
195
196 /* Case à cocher */
197 SDL_FRect case_rect = { panneau_x + PADDING, y + 5.0f, 14.0f, 14.0f };
198 SDL_SetRenderDrawColor(renderer, 80, 130, 200, 255);
199 SDL_RenderRect(renderer, &case_rect);
200
201 if (obj->objectifs[i].valide) {
202 /* Croix verte intérieure */
203 SDL_SetRenderDrawColor(renderer, 50, 220, 80, 255);
204 SDL_RenderLine(renderer,
205 case_rect.x + 2, case_rect.y + 2,
206 case_rect.x + 12, case_rect.y + 12);
207 SDL_RenderLine(renderer,
208 case_rect.x + 12, case_rect.y + 2,
209 case_rect.x + 2, case_rect.y + 12);
210 }
211
212 /* Texte de l'objectif */
213 SDL_Color couleur;
214 if (obj->objectifs[i].valide) {
215 couleur = (SDL_Color){100, 220, 100, 255}; /* vert validé */
216 } else {
217 couleur = (SDL_Color){200, 200, 200, 255}; /* gris en attente */
218 }
219
220 SDL_Surface *s = TTF_RenderText_Solid(font, obj->objectifs[i].description, 0, couleur);
221 if (s) {
222 float max_w = PANNEAU_W - PADDING * 2 - 20.0f;
223 float draw_w = (s->w > max_w) ? max_w : (float)s->w;
224 SDL_FRect dst_txt = { panneau_x + PADDING + 20.0f, y + 2.0f, draw_w, (float)s->h };
225
226 SDL_Texture *t = SDL_CreateTextureFromSurface(renderer, s);
227 SDL_FRect full_src = { 0, 0, (float)s->w, (float)s->h };
228 SDL_RenderTexture(renderer, t, &full_src, &dst_txt);
229 SDL_DestroyTexture(t);
230 SDL_DestroySurface(s);
231 }
232
233 /* Trait de barrage sur texte validé */
234 if (obj->objectifs[i].valide) {
235 SDL_SetRenderDrawColor(renderer, 80, 180, 80, 200);
236 float mid_y = y + LIGNE_H / 2.5f;
237 SDL_RenderLine(renderer,
238 panneau_x + PADDING + 20.0f, mid_y,
239 panneau_x + PANNEAU_W - PADDING, mid_y);
240 }
241
242 y += LIGNE_H;
243 }
244
245 /* Compteur de progression en bas */
246 int valides = 0;
247 for (int i = 0; i < obj->nb; i++) if (obj->objectifs[i].valide) valides++;
248
249 char prog[32];
250 SDL_snprintf(prog, sizeof(prog), "%d / %d complétés", valides, obj->nb);
251 SDL_Color gris = {140, 150, 170, 255};
252 SDL_Surface *s_prog = TTF_RenderText_Solid(font, prog, 0, gris);
253 if (s_prog) {
254 SDL_Texture *t = SDL_CreateTextureFromSurface(renderer, s_prog);
255 SDL_FRect dst = {
256 panneau_x + (PANNEAU_W - s_prog->w) / 2.0f,
257 panneau_y + hauteur - s_prog->h - PADDING / 2.0f,
258 (float)s_prog->w, (float)s_prog->h
259 };
260 SDL_RenderTexture(renderer, t, NULL, &dst);
261 SDL_DestroyTexture(t);
262 SDL_DestroySurface(s_prog);
263 }
264}
