28int Bouton_GererEvenement(Bouton *b, const SDL_Event *e) {
29 int clique = 0;
30
31 if (e->type == SDL_EVENT_MOUSE_MOTION) {
32 float mx = e->motion.x;
33 float my = e->motion.y;
34 b->survole = pointDansRect(mx, my, &b->rect);
35
36 } else if (e->type == SDL_EVENT_MOUSE_BUTTON_DOWN && e->button.button == SDL_BUTTON_LEFT) {
37 float mx = e->button.x;
38 float my = e->button.y;
39 if (pointDansRect(mx, my, &b->rect)) {
40 b->appuye = 1;
41 }
42
43 } else if (e->type == SDL_EVENT_MOUSE_BUTTON_UP && e->button.button == SDL_BUTTON_LEFT) {
44 float mx = e->button.x;
45 float my = e->button.y;
46 if (b->appuye && pointDansRect(mx, my, &b->rect)) {
47 clique = 1;
48 }
49 b->appuye = 0;
50 }
51
52 return clique;
53}
