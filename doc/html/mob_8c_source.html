46static int pop_respawn_prevu(EntreeRespawn *out) {
47 if (tete_file == queue_file) return -1;
48 Uint32 maintenant = SDL_GetTicks();

50 if (out) *out = file_respawn[tete_file];

218/* Spawn a single mob of given type id at a random valid ground tile anywhere on the map.
219 Returns 0 on success, -1 on failure. */
220static int spawn_one_mob_of_type(Mob * mobs[MAX_MOB], t_tile map[W_MAP][H_MAP], int type_id) {
221 SDL_FRect rect_vaisseau = get_vaisseau_collision_rect();
222 /* try a number of random attempts */
223 for (int attempt = 0; attempt < 200; attempt++) {
224 int tx = rand() % W_MAP;
225 int ty = rand() % H_MAP;
226 type_t t = map[tx][ty].type;
227 int est_terre = (t == terreP || t == terreCHG || t == terreCHD || t == terreCBG || t == terreCBD || t == terreH || t == terreB || t == terreG || t == terreD);
228 if (!est_terre) continue;
229 SDL_Rect r = { tx * DISPLAY_TILE_SIZE, ty * DISPLAY_TILE_SIZE, DISPLAY_TILE_SIZE, DISPLAY_TILE_SIZE };
230 if (test_collision(tx, ty, map, 1, r)) continue;
231 SDL_FRect r_f = {(float)r.x, (float)r.y, (float)r.w, (float)r.h};
232 if (SDL_HasRectIntersectionFloat(&r_f, &rect_vaisseau)) continue;
233
234 /* find empty slot in mobs[] */
235 int idx = 0;
236 while (idx < MAX_MOB && mobs[idx] != NULL) idx++;
237 if (idx >= MAX_MOB) return -1;
238
239 mobs[idx] = malloc(sizeof(Mob));
240 mobs[idx]->largeur = 1; mobs[idx]->hauteur = 1;
241 int jitter = DISPLAY_TILE_SIZE / 4;
242 int ox = (rand() % (jitter * 2 + 1)) - jitter;
243 int oy = (rand() % (jitter * 2 + 1)) - jitter;
244 mobs[idx]->x = tx * DISPLAY_TILE_SIZE + (DISPLAY_TILE_SIZE/2) + ox - (mobs[idx]->largeur*DISPLAY_TILE_SIZE)/2;
245 mobs[idx]->y = ty * DISPLAY_TILE_SIZE + (DISPLAY_TILE_SIZE/2) + oy - (mobs[idx]->hauteur*DISPLAY_TILE_SIZE)/2;
246 mobs[idx]->direction = 0;
247 mobs[idx]->vx = 0.0f; mobs[idx]->vy = 0.0f;
248 mobs[idx]->target_vx = 0.0f; mobs[idx]->target_vy = 0.0f;
249 mobs[idx]->time_change_dir = 0;
250 mobs[idx]->vie = 3;
251 mobs[idx]->type_boss_proprietaire = -1;
252 mobs[idx]->drop_chance = 100;
253 if (type_id == 1) { mobs[idx]->speed = 30.0f; mobs[idx]->texture = texture_mouton; mobs[idx]->id = 1; }
254 else { mobs[idx]->speed = 70.0f; mobs[idx]->texture = texture_pawns; mobs[idx]->id = 2; }
255 return 0;
256 }

260void update_mobs(t_tile map[W_MAP][H_MAP], Mob * mobs[MAX_MOB]) {
261 static Uint32 last_time = 0;
262 Uint32 now = SDL_GetTicks();
263 /* process due respawns */
264 EntreeRespawn entree;
265 while (pop_respawn_prevu(&entree) == 0) {
266 spawn_one_mob_of_type(mobs, map, entree.id);
267 }
268 if (last_time == 0) last_time = now;
269 float dt = (now - last_time) / 1000.0f; /* seconds */
270 if (dt <= 0) dt = 0.001f;
271 SDL_FRect rect_vaisseau = get_vaisseau_collision_rect();
272
273 for (int i = 0; mobs[i] != NULL; i++) {
274 Mob *m = mobs[i];
275 Uint32 timer = now;
276
277 if (m->id == 3) {
278 float joueur_x = 0.0f;
279 float joueur_y = 0.0f;
280 get_player_world_center(&joueur_x, &joueur_y);
281 float minion_x = m->x + (m->largeur * DISPLAY_TILE_SIZE * 0.5f);
282 float minion_y = m->y + (m->hauteur * DISPLAY_TILE_SIZE * 0.5f);
283 float dx = joueur_x - minion_x;
284 float dy = joueur_y - minion_y;
285 float dist = sqrtf(dx * dx + dy * dy);
286
287 if (dist > 95.0f) {
288 if (dist < 0.001f) dist = 1.0f;
289 m->target_vx = (dx / dist) * m->speed;
290 m->target_vy = (dy / dist) * m->speed;
291 } else {
292 m->target_vx = 0.0f;
293 m->target_vy = 0.0f;
294
295 if (perso.vie > 0 && perso.invincibiliter_timer == 0 && (timer - minion_attack_timer) > 850) {
296 perso.vie -= 1;
297 if (perso.vie < 0) perso.vie = 0;
298 perso.invincibiliter_timer = SDL_GetTicks();
299 minion_attack_timer = timer;
300 }
301 }
302 m->time_change_dir = timer + 120;
303 } else if (timer >= m->time_change_dir) {
304 float angle = (rand() % 360) * (3.14159265f / 180.0f);
305 int change_ms = (m->id == 1) ? 2000 + (rand() % 2000) : 800 + (rand() % 1200);
306 m->target_vx = cosf(angle) * m->speed;
307 m->target_vy = sinf(angle) * m->speed;
308 m->time_change_dir = timer + change_ms;
309 }
310
311 float steer_strength = (m->id == 1) ? 200.0f : ((m->id == 3) ? 650.0f : 400.0f); /* px/s^2 */
312 float ax = (m->target_vx - m->vx) * fminf(1.0f, steer_strength * dt / fmaxf(1.0f, m->speed));
313 float ay = (m->target_vy - m->vy) * fminf(1.0f, steer_strength * dt / fmaxf(1.0f, m->speed));
314 m->vx += ax;
315 m->vy += ay;
316
317 float vmag = sqrtf(m->vx * m->vx + m->vy * m->vy);
318 if (vmag > m->speed) {
319 m->vx = (m->vx / vmag) * m->speed;
320 m->vy = (m->vy / vmag) * m->speed;
321 }
322
323 float nouvelle_x = m->x + m->vx * dt;
324 float nouvelle_y = m->y + m->vy * dt;
325
326 SDL_Rect proj = {
327 (int)nouvelle_x,
328 (int)nouvelle_y,
329 (int)(m->largeur * DISPLAY_TILE_SIZE),
330 (int)(m->hauteur * DISPLAY_TILE_SIZE)
331 };
332 SDL_FRect proj_f = {(float)proj.x, (float)proj.y, (float)proj.w, (float)proj.h};
333
334 int tile_x = (int)((nouvelle_x + m->largeur * DISPLAY_TILE_SIZE / 2) / DISPLAY_TILE_SIZE);
335 int tile_y = (int)((nouvelle_y + m->hauteur * DISPLAY_TILE_SIZE / 2) / DISPLAY_TILE_SIZE);
336
337 if (tile_x >= 0 && tile_x < W_MAP && tile_y >= 0 && tile_y < H_MAP) {
338 if (!test_collision(tile_x, tile_y, map, 1, proj) && !SDL_HasRectIntersectionFloat(&proj_f, &rect_vaisseau)) {
339 m->x = nouvelle_x;
340 m->y = nouvelle_y;
341 } else {
342 if (m->id != 3) {
343 float angle = (rand() % 360) * (3.14159265f / 180.0f);
344 m->target_vx = cosf(angle) * m->speed;
345 m->target_vy = sinf(angle) * m->speed;
346 m->vx *= 0.3f;
347 m->vy *= 0.3f;
348 m->time_change_dir = timer + 300 + (rand() % 400);
349 } else {
350 SDL_Rect proj_x = {
351 (int)nouvelle_x,
352 (int)m->y,
353 (int)(m->largeur * DISPLAY_TILE_SIZE),
354 (int)(m->hauteur * DISPLAY_TILE_SIZE)
355 };
356 SDL_FRect proj_x_f = {(float)proj_x.x, (float)proj_x.y, (float)proj_x.w, (float)proj_x.h};
357 int tile_x_only = (int)((nouvelle_x + m->largeur * DISPLAY_TILE_SIZE / 2) / DISPLAY_TILE_SIZE);
358
359 SDL_Rect proj_y = {
360 (int)m->x,
361 (int)nouvelle_y,
362 (int)(m->largeur * DISPLAY_TILE_SIZE),
363 (int)(m->hauteur * DISPLAY_TILE_SIZE)
364 };
365 SDL_FRect proj_y_f = {(float)proj_y.x, (float)proj_y.y, (float)proj_y.w, (float)proj_y.h};
366 int tile_y_only = (int)((nouvelle_y + m->hauteur * DISPLAY_TILE_SIZE / 2) / DISPLAY_TILE_SIZE);
367
368 int move_x_ok = (tile_x_only >= 0 && tile_x_only < W_MAP) &&
369 !test_collision(tile_x_only, (int)((m->y + m->hauteur * DISPLAY_TILE_SIZE / 2) / DISPLAY_TILE_SIZE), map, 1, proj_x) &&
370 !SDL_HasRectIntersectionFloat(&proj_x_f, &rect_vaisseau);
371 int move_y_ok = (tile_y_only >= 0 && tile_y_only < H_MAP) &&
372 !test_collision((int)((m->x + m->largeur * DISPLAY_TILE_SIZE / 2) / DISPLAY_TILE_SIZE), tile_y_only, map, 1, proj_y) &&
373 !SDL_HasRectIntersectionFloat(&proj_y_f, &rect_vaisseau);
374
375 if (move_x_ok) {
376 m->x = nouvelle_x;
377 m->vy *= 0.5f;
378 }
379 if (move_y_ok) {
380 m->y = nouvelle_y;
381 m->vx *= 0.5f;
382 }
383
384 if (!move_x_ok && !move_y_ok) {
385 float joueur_x = 0.0f;
386 float joueur_y = 0.0f;
387 float minion_x = m->x + (m->largeur * DISPLAY_TILE_SIZE * 0.5f);
388 float minion_y = m->y + (m->hauteur * DISPLAY_TILE_SIZE * 0.5f);
389 get_player_world_center(&joueur_x, &joueur_y);
390 float dx = joueur_x - minion_x;
391 float dy = joueur_y - minion_y;
392 float dist = sqrtf(dx * dx + dy * dy);
393 if (dist < 0.001f) dist = 1.0f;
394 float nx = dx / dist;
395 float ny = dy / dist;
396 float s = (rand() % 2 == 0) ? 1.0f : -1.0f;
397 m->target_vx = -ny * m->speed * s;
398 m->target_vy = nx * m->speed * s;
399 m->vx *= 0.3f;
400 m->vy *= 0.3f;
401 }
402 }
403 }
404 } else {
405 float center_x = (W_MAP * DISPLAY_TILE_SIZE) / 2.0f;
406 float center_y = (H_MAP * DISPLAY_TILE_SIZE) / 2.0f;
407 float dx = center_x - m->x;
408 float dy = center_y - m->y;
409 float ang = atan2f(dy, dx);
410 m->target_vx = cosf(ang) * m->speed;
411 m->target_vy = sinf(ang) * m->speed;
412 m->vx *= 0.5f;
413 m->vy *= 0.5f;
414 m->time_change_dir = timer + 500;
415 }
416 }
417
418 last_time = now;
419}

421void afficher_mob(SDL_Renderer * renderer, Mob * mobs[MAX_MOB]){
422 for(int i = 0; mobs[i] != NULL; i++){
423 if (mobs[i]->texture == NULL) {
424 SDL_Log("erreur mob : texture invalide (id=%d, index=%d)", mobs[i]->id, i);
425 continue;
426 }
427 SDL_FRect dest = {
428 .x = mobs[i]->x + perso.x,
429 .y = mobs[i]->y + perso.y,
430 .w = mobs[i]->largeur * DISPLAY_TILE_SIZE,
431 .h = mobs[i]->hauteur * DISPLAY_TILE_SIZE
432 };
433 SDL_FRect src_pawn = {
434 .x = 0,
435 .y = 0,
436 .w = 192,
437 .h = 192
438 };
439 SDL_FRect src_mouton = {
440 .x = 0,
441 .y = 0,
442 .w = 128,
443 .h = 128
444 };
445 if (mobs[i]->id == 3 && (texture_minion_run != NULL || texture_minion_attack != NULL)) {
446 float joueur_x = 0.0f;
447 float joueur_y = 0.0f;
448 get_player_world_center(&joueur_x, &joueur_y);
449 float minion_x = mobs[i]->x + (mobs[i]->largeur * DISPLAY_TILE_SIZE * 0.5f);
450 float minion_y = mobs[i]->y + (mobs[i]->hauteur * DISPLAY_TILE_SIZE * 0.5f);
451 float dx = joueur_x - minion_x;
452 float dy = joueur_y - minion_y;
453 float dist = sqrtf(dx * dx + dy * dy);
454
455 int is_attack = (dist <= 100.0f);
456 SDL_Texture *tex = is_attack ? texture_minion_attack : texture_minion_run;
457 int frames = is_attack ? 4 : 6;
458 if (tex == NULL) {
459 tex = texture_minion_run;
460 frames = 6;
461 }
462 if (tex == NULL) {
463 tex = mobs[i]->texture;
464 frames = 1;
465 }
466
467 float tw = 192.0f;
468 float th = 192.0f;
469 SDL_GetTextureSize(tex, &tw, &th);
470
471 int frame = (int)((SDL_GetTicks() / 110) % (Uint32)frames);
472 SDL_FRect src_minion = {
473 .x = (tw / (float)frames) * (float)frame,
474 .y = 0,
475 .w = (tw / (float)frames),
476 .h = th
477 };
478 SDL_RenderTexture(renderer, tex, &src_minion, &dest);
479 } else {
480 if (mobs[i]->texture == NULL) SDL_Log("erreur mob : %s", SDL_GetError());
481 SDL_RenderTexture(renderer, mobs[i]->texture, mobs[i]->texture == texture_pawns ? &src_pawn : &src_mouton, &dest);
482 }
