10static SDL_Texture *make_text(SDL_Renderer *renderer, TTF_Font *font, const char *txt, SDL_Color color) {
11 SDL_Surface *s = TTF_RenderText_Solid(font, txt, SDL_strlen(txt), color);
12 if (!s) return NULL;
13 SDL_Texture *t = SDL_CreateTextureFromSurface(renderer, s);
14 SDL_DestroySurface(s);
15 return t;
16}

18static void draw_line(SDL_Renderer *renderer, TTF_Font *font, const char *txt, float x, float y, SDL_Color color) {
19 SDL_Texture *t = make_text(renderer, font, txt, color);
20 if (!t) return;
21 float tw = 0.0f, th = 0.0f;
22 SDL_GetTextureSize(t, &tw, &th);
23 SDL_FRect dst = {x, y, tw, th};
24 SDL_RenderTexture(renderer, t, NULL, &dst);
25 SDL_DestroyTexture(t);
26}
