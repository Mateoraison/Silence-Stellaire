82static uint32_t calculer_checksum_save_v3(const SaveData *data) {
83 SaveData tmp = *data;
84 tmp.checksum = 0;
85
86 const unsigned char *bytes = (const unsigned char *)&tmp;
87 uint32_t hash = 2166136261u;
88 for (size_t i = 0; i < sizeof(SaveData); i++) {
89 hash ^= (uint32_t)bytes[i];
90 hash *= 16777619u;
91 }
92 return hash;
93}

105static int save_data_valide_v3(const SaveData *data) {
106 if (data->magic != SAVE_MAGIC) return 0;
107 if (data->version != SAVE_VERSION) return 0;
108 if (data->planete < 1 || data->planete > 3) return 0;
109 if (data->vie_max <= 0 || data->faim_max <= 0) return 0;
110 if (data->vie < 0 || data->vie > data->vie_max) return 0;
111 if (data->faim < 0 || data->faim > data->faim_max) return 0;
112 if (data->direction < 0 || data->direction > 3) return 0;
113
114 for (int i = 0; i < HOTBAR_SIZE; i++) {
115 if (data->hotbar_qte[i] < 0) return 0;
116 }
117 for (int i = 0; i < INVENTAIRE_SIZE; i++) {
118 if (data->inventaire_qte[i] < 0) return 0;
119 }
120 for (int i = 0; i < CAISSE_OUTILS_SIZE; i++) {
121 if (data->caisse_qte[i] < 0) return 0;
122 }
123
124 if (!save_checksum_ok_v3(data)) return 0;
125 return 1;
126}

141static void reconstruire_case_array(t_case **cases, int taille, const int *types, const int *qtes, SDL_Renderer *renderer) {
142 for (int i = 0; i < taille; i++) {
143 if (types[i] < 0 || qtes[i] <= 0) continue;
144
145 t_Item *base = init_item((typeItem)types[i], renderer, 0.0f, 0.0f);
146 if (!base) continue;
147
148 cases[i] = malloc(sizeof(t_case));
149 if (!cases[i]) {
150 free(base);
151 continue;
152 }
153
154 cases[i]->item = malloc(sizeof(t_Item));
155 if (!cases[i]->item) {
156 free(cases[i]);
157 cases[i] = NULL;
158 free(base);
159 continue;
160 }
161
162 cases[i]->item->type = (typeItem)types[i];
163 cases[i]->item->texture = base->texture;
164 cases[i]->item->x = 0;
165 cases[i]->item->y = 0;
166 cases[i]->quantiter = qtes[i];
167 cases[i]->stackable = 1;
168 cases[i]->x = 0;
169 cases[i]->y = 0;
170 cases[i]->w = 0;
171 cases[i]->h = 0;
172
173 free(base);
174 }

306int charger_partie_slot(int slot, int *planete_out) {
307 if (slot < 1 || slot > NB_SLOTS_SAUVEGARDE) return -1;
308
309 char path[64];
310 chemin_slot(slot, path);
311 FILE *f = fopen(path, "rb");
312 if (!f) return -1;
313
314 fseek(f, 0, SEEK_END);
315 long file_size = ftell(f);
316 if (file_size < 0) {
317 fclose(f);
318 return -1;
319 }
320 rewind(f);
321
322 SaveData data;
323 SDL_memset(&data, 0, sizeof(data));
324
325 if ((size_t)file_size == sizeof(SaveData)) {
326 size_t n = fread(&data, sizeof(SaveData), 1, f);
327 fclose(f);
328 if (n != 1) return -1;
329 if (!save_data_valide_v3(&data)) return -1;
330 } else {
331 fclose(f);
332 return -1;
333 }
334
335 sauvegarde_attente = data;
336 chargement_en_attente = 1;
337 slot_actif = slot;
338 if (planete_out) *planete_out = data.planete;
339 return 0;
340}
