59static void liberer_case_array(t_case **cases, int taille) {
60 for (int i = 0; i < taille; i++) {
61 if (cases[i]) {
62 if (cases[i]->item) {
63 free(cases[i]->item);
64 }
65 free(cases[i]);
66 cases[i] = NULL;
67 }
68 }

71static void reconstruire_case_array(t_case **cases, int taille, const int *types, const int *qtes, SDL_Renderer *renderer) {
72 for (int i = 0; i < taille; i++) {
73 if (types[i] < 0 || qtes[i] <= 0) continue;
74
75 t_Item *base = init_item((typeItem)types[i], renderer, 0.0f, 0.0f);
76 if (!base) continue;
77
78 cases[i] = malloc(sizeof(t_case));
79 if (!cases[i]) {
80 free(base);
81 continue;
82 }
83
84 cases[i]->item = malloc(sizeof(t_Item));
85 if (!cases[i]->item) {
86 free(cases[i]);
87 cases[i] = NULL;
88 free(base);
89 continue;
90 }
91
92 cases[i]->item->type = (typeItem)types[i];
93 cases[i]->item->texture = base->texture;
94 cases[i]->item->x = 0;
95 cases[i]->item->y = 0;
96 cases[i]->quantiter = qtes[i];
97 cases[i]->stackable = 1;
98 cases[i]->x = 0;
99 cases[i]->y = 0;
100 cases[i]->w = 0;
101 cases[i]->h = 0;
102
103 free(base);
104 }
105}

147int sauvegarder_partie_slot(int slot, int planete) {
148 if (slot < 1 || slot > NB_SLOTS_SAUVEGARDE) return -1;
149
150 SaveData data;

152 data.version = SAVE_VERSION;

155 data.perso_y = perso.y;
156 data.direction = perso.direction;
157 data.vie = perso.vie;
158 data.vie_max = perso.vie_max;
159 data.faim = perso.faim;

163 data.engrenages_poses = engrenages_poses;
164 data.vaisseau_repare = vaisseau_repare ? 1 : 0;
165
166 for (int i = 0; i < HOTBAR_SIZE; i++) {
167 data.hotbar_type[i] = -1;
168 data.hotbar_qte[i] = 0;
169 if (hotbar[i] && hotbar[i]->item) {
170 data.hotbar_type[i] = hotbar[i]->item->type;
171 data.hotbar_qte[i] = hotbar[i]->quantiter;
172 }
173 }
174
175 for (int i = 0; i < INVENTAIRE_SIZE; i++) {
176 data.inventaire_type[i] = -1;
177 data.inventaire_qte[i] = 0;
178 if (inventaire[i] && inventaire[i]->item) {
179 data.inventaire_type[i] = inventaire[i]->item->type;
180 data.inventaire_qte[i] = inventaire[i]->quantiter;
181 }
182 }
183
184 for (int i = 0; i < CAISSE_OUTILS_SIZE; i++) {
185 data.caisse_type[i] = -1;
186 data.caisse_qte[i] = 0;
187 if (caisse_outils[i] && caisse_outils[i]->item) {
188 data.caisse_type[i] = caisse_outils[i]->item->type;
189 data.caisse_qte[i] = caisse_outils[i]->quantiter;
190 }
191 }
192
193 assurer_dossier_saves();
194 char path[64];
195 chemin_slot(slot, path);
196
197 FILE *f = fopen(path, "wb");
198 if (!f) return -1;
199 size_t n = fwrite(&data, sizeof(SaveData), 1, f);
200 fclose(f);
201 if (n != 1) return -1;
202
203 slot_actif = slot;
204 return 0;
205}

207int charger_partie_slot(int slot, int *planete_out) {
208 if (slot < 1 || slot > NB_SLOTS_SAUVEGARDE) return -1;
209
210 char path[64];
211 chemin_slot(slot, path);
212 FILE *f = fopen(path, "rb");
213 if (!f) return -1;
214
215 SaveData data;
216 size_t n = fread(&data, sizeof(SaveData), 1, f);
217 fclose(f);
218 if (n != 1) return -1;
219 if (data.magic != SAVE_MAGIC || data.version != SAVE_VERSION) return -1;
220
221 sauvegarde_attente = data;
222 chargement_en_attente = 1;
223 slot_actif = slot;
224 if (planete_out) *planete_out = data.planete;
225 return 0;

254int sauvegarde_choisir_slot(SDL_Renderer *renderer, const char *titre, bool mode_chargement) {
255 TTF_Font *font = TTF_OpenFont("assets/police.ttf", 24);
256 if (!font) return 0;
257
258 SDL_Texture *fond = IMG_LoadTexture(renderer, "assets/menu/menu.png");
259 SDL_Texture *btn_tex = IMG_LoadTexture(renderer, "assets/menu/bouton.png");
260 if (!btn_tex) {
261 if (fond) SDL_DestroyTexture(fond);
262 TTF_CloseFont(font);
263 return 0;
264 }
265
266 Bouton slots[NB_SLOTS_SAUVEGARDE];
267 float base_x = screen_center_x() - 170.0f;
268 float base_y = screen_center_y() - ((NB_SLOTS_SAUVEGARDE + 1) * 95.0f) * 0.5f;
269 for (int i = 0; i < NB_SLOTS_SAUVEGARDE; i++) {
270 Bouton_Init(&slots[i], base_x, base_y + i * 95.0f, 340.0f, 72.0f, btn_tex);
271 }
272 Bouton btn_retour;
273 Bouton_Init(&btn_retour, base_x, base_y + NB_SLOTS_SAUVEGARDE * 95.0f, 340.0f, 72.0f, btn_tex);
274
275 SDL_Color blanc = {255, 255, 255, 255};
276 SDL_Color gris = {180, 180, 180, 255};
277 SDL_Color jaune = {255, 220, 80, 255};
278
279 SDL_Texture *tex_titre = creer_texte(renderer, font, titre, blanc);
280
281 SDL_Texture *tex_slots[NB_SLOTS_SAUVEGARDE] = {NULL};
282 for (int i = 0; i < NB_SLOTS_SAUVEGARDE; i++) {
283 char txt[64];
284 int occupe = sauvegarde_slot_existe(i + 1);
285 SDL_snprintf(txt, sizeof(txt), "Slot %d - %s", i + 1, occupe ? "Occupe" : "Vide");
286 tex_slots[i] = creer_texte(renderer, font, txt, occupe ? blanc : gris);
287 }
288 SDL_Texture *tex_retour = creer_texte(renderer, font, "Retour", blanc);
289
290 char notif[96] = "";
291 Uint32 notif_debut = 0;
292
293 int slot_choisi = 0;
294 bool running = true;
295 while (running) {
296 update_screen_metrics(renderer);
297 base_x = screen_center_x() - 170.0f;
298 base_y = screen_center_y() - ((NB_SLOTS_SAUVEGARDE + 1) * 95.0f) * 0.5f;
299 for (int i = 0; i < NB_SLOTS_SAUVEGARDE; i++) {
300 slots[i].rect.x = base_x;
301 slots[i].rect.y = base_y + i * 95.0f;
302 }
303 btn_retour.rect.x = base_x;
304 btn_retour.rect.y = base_y + NB_SLOTS_SAUVEGARDE * 95.0f;
305
306 SDL_Event ev;
307 while (SDL_PollEvent(&ev)) {
308 if (ev.type == SDL_EVENT_MOUSE_MOTION ||
309 ev.type == SDL_EVENT_MOUSE_BUTTON_DOWN ||
310 ev.type == SDL_EVENT_MOUSE_BUTTON_UP) {
311 SDL_ConvertEventToRenderCoordinates(renderer, &ev);
312 }
313
314 if (ev.type == SDL_EVENT_QUIT) { running = false; break; }
315 if (ev.type == SDL_EVENT_KEY_DOWN && ev.key.key == SDLK_ESCAPE) { running = false; break; }
316
317 if (Bouton_GererEvenement(&btn_retour, &ev)) {
318 running = false;
319 break;
320 }
321
322 for (int i = 0; i < NB_SLOTS_SAUVEGARDE; i++) {
323 if (Bouton_GererEvenement(&slots[i], &ev)) {
324 int slot = i + 1;
325 if (mode_chargement && !sauvegarde_slot_existe(slot)) {
326 SDL_snprintf(notif, sizeof(notif), "Le slot %d est vide", slot);
327 notif_debut = SDL_GetTicks();
328 } else {
329 slot_choisi = slot;
330 running = false;
331 }
332 break;
333 }
334 }
335 }
336
337 SDL_RenderClear(renderer);
338 if (fond) SDL_RenderTexture(renderer, fond, NULL, NULL);
339
340 if (tex_titre) {
341 SDL_FRect r = {screen_center_x() - 170.0f, base_y - 70.0f, 340.0f, 40.0f};
342 SDL_RenderTexture(renderer, tex_titre, NULL, &r);
343 }
344
345 for (int i = 0; i < NB_SLOTS_SAUVEGARDE; i++) {
346 Bouton_Afficher(&slots[i], renderer);
347 if (tex_slots[i]) afficher_texte_centre(renderer, tex_slots[i], &slots[i].rect);
348 }
349 Bouton_Afficher(&btn_retour, renderer);
350 if (tex_retour) afficher_texte_centre(renderer, tex_retour, &btn_retour.rect);
351
352 if (notif[0] != '\0' && SDL_GetTicks() - notif_debut < 1800) {
353 SDL_Texture *tex_notif = creer_texte(renderer, font, notif, jaune);
354 if (tex_notif) {
355 SDL_FRect r = {screen_center_x() - 140.0f, btn_retour.rect.y + btn_retour.rect.h + 14.0f, 280.0f, 30.0f};
356 SDL_RenderTexture(renderer, tex_notif, NULL, &r);
357 SDL_DestroyTexture(tex_notif);
358 }
359 }
360
361 SDL_RenderPresent(renderer);
362 SDL_Delay(16);
363 }
364
365 if (tex_titre) SDL_DestroyTexture(tex_titre);
366 for (int i = 0; i < NB_SLOTS_SAUVEGARDE; i++) {
367 if (tex_slots[i]) SDL_DestroyTexture(tex_slots[i]);
368 }
369 if (tex_retour) SDL_DestroyTexture(tex_retour);
370 if (fond) SDL_DestroyTexture(fond);
371 SDL_DestroyTexture(btn_tex);
372 TTF_CloseFont(font);
373
374 return slot_choisi;
375}
